7 /** Constructor for charged particle reactions. Specifies the number of
8  * resonances as well as the charge and reduced mass of the reactants.
9  *
10  * For charged particle reactions the non-resonant set of parameters has a0
11  * fixed based on S(0) (See ReaclibRate::SetSFactor). The a1 term is set based
12  * on the charge and mass of the reactants in the following form:
13  * \f[
14  * -4.2486 (Z_1^2 Z_2^2 \mu)^{1/3}
15  * \f]
16  * where \f$ Z_1 \f$ and \f$ Z_2 \f$ are the charge and \f$ \mu \f$ of the
17  * reaction reactants.
18  * The a3 through a5 terms are allowed to float and the a6 term is set to -2/3.
19  *
20  * For narrow resonances the a0 and a1 term are set based on the resonance
21  * energy and strength (See ReaclibRate::SetResonance). The a2 through a5
22  * terms are fixed to 0. Finally, a6 is set to -3/2.
23  *
24  * \note Neutron induced non-resonant reaction rates are not yet supported.
25  */
26 ReaclibRate::ReaclibRate(
27  const char* name, const unsigned int numResonances,
28  const unsigned int z1, const unsigned int z2, const float mu
29 ) :
30  TF1(name, this, &ReaclibRate::Evaluate, 0.01, 10, 7 * (numResonances+1)),
31  numResonances_(numResonances),
32  z1_(z1),
33  z2_(z2),
34  mu_amu_(mu)
35 {
36  //First set the non-resonant set of terms.

42  FixParameter(6, -2./3.);

57 /** Sets the term (a0) of the non-resonant set associated with the s-factor at
58  * energy zero, S(0). The a0 term takes the form
59  * \f[
60  * ln[B (Z_1 Z_2 \mu)^{1/3} S(0)]
61  * \f]
62  * where \f$ B = 7.8318 \times 10^9 cm^3 s^{-1} mole^{-1} MeV^{-1} \f$,
63  * \f$ Z_1 \f$ and \f$ Z_2 \f$ are the charge of the reactants, \f$ \mu \f$ is
64  * the reduced mass of the reactants and \f$ S(0) \f$ is the value of the
65  * S-factor evaluated at an energy of zero.
66  *
67  * The parameter is fixed, but can be allowed to float by
68  * using the following: @code ReaclibRate::SetParLimits(0, 0, 0); @endcode
69  */
70 void ReaclibRate::SetSFactor(float s0_MeVb) {
71  FixParameter(0, log(b_ * pow(z1_ * z2_ * mu_amu_, 1./3.) * s0_MeVb));

74 /**Sets the terms for a resonance set. Specifically, a0 and a1 are set using
75  * the resonance strength and energy.
76  *
77  * For narrow resonances the a0 term takes on the following form:
78  * \f[
79  * ln[D \mu^{-3/2} \omega\gamma]
80  * \f]
81  * where \f$ D = 1.5394 \times 10^{11} cm^3 s^{-1} mole^{-1} MeV^{-1} \f$,
82  * \f$ \mu \f$ is the reactants reduced mass, and \f$ \omega \gamma \f$ is the
83  * narrow resonance strength.
84  *
85  * The a1 term takes on the following form:
86  * \f[
87  * -11.6045 E_r
88  * \f]
89  * where \f$ E_r \f$ is the resonance energy.
90  *
91  * The corresponding parameters are fixed, but can be allowed to float by
92  * using the following:
93  * @code
94  * ReaclibRate::SetParLimits(7 * (resonanceId + 1) + 0, 0, 0);
95  * ReaclibRate::SetParLimits(7 * (resonanceId + 1) + 1, 0, 0);
96  * @endcode
97  */
98 void ReaclibRate::SetResonance(
99  const unsigned int resonanceId, const float energy, const float strength
100 ) {
101  if (resonanceId <= numResonances_) {
102  FixParameter(7 * (resonanceId+1) + 0, log(d_ * pow(mu_amu_, -3./2.) * strength));
103  FixParameter(7 * (resonanceId+1) + 1, -11.6045 * energy);
104  }

139 /**Evaluates the reaction by summing each set. The zeroth set is the
140  * non-resonant term while every additional set are resonant contributions.
141  * The terms are evaluated using the following equation:
142  * @f[
143  * \sum_{n}exp\left[
144  * a_{n,0} +\sum_{i=1}^5 a_{n,i} T_9^{2i-5/3} + a_{n,6} \ln T_9
145  * \right]
146  * @f]
147  * where @f$ n @f$ is the set index. Typically a rate has one set for the
148  * non-resonant contribution and all subsequent sets are from narrow resonances.
149  */
150 double ReaclibRate::Evaluate(double *t9, double *par) {
151  //Value of the reaction rate at the specified t9 value.
152  double reacRate = 0;
153  //Loop over the resonances plus one for the non-resonant contribution.
154  for (unsigned int i=0; i< numResonances_ + 1; i++) {
155  //Compute the contribution from this component.
156  // We apply the exponential at the end of each set.
157  double component = par[7*i+0] + par[7*i+6] * log(t9[0]);
158  for (int j=1;j<=5;j++) {
159  component += par[7*i+j] * pow(t9[0],(2.*j-5.)/3.);
160  }
161  //Add the contribution to the total rate.
162  reacRate += exp(component);
163  }
164  return reacRate;
165 }
